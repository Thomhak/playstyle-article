# Function to plot session timeline for a player
plot_player_sessions <- function(player_data, player_id = NULL, player_label = "Player") {
  # If a player_id is provided, filter data for that player
  if (!is.null(player_id)) {
    player_data <- player_data %>%
      filter(player_id == !!player_id)
  }
  
  # Check if the data already has formatted time columns
  if (!("start_time_only" %in% names(player_data)) || !("end_time_only" %in% names(player_data))) {
    # Convert timestamps to POSIXct if they're not already
    if (!inherits(player_data$game_start_time, "POSIXct")) {
      player_data$game_start_time <- as.POSIXct(player_data$game_start_time, format = "%Y-%m-%d %H:%M:%S")
      player_data$game_end_time <- as.POSIXct(player_data$game_end_time, format = "%Y-%m-%d %H:%M:%S")
    }
    
    # Extract time component only
    player_data$start_time_only <- lubridate::as_datetime(
      format(player_data$game_start_time, format = "%H:%M:%S"), 
      format = "%H:%M:%S"
    )
    player_data$end_time_only <- lubridate::as_datetime(
      format(player_data$game_end_time, format = "%H:%M:%S"), 
      format = "%H:%M:%S"
    )
  }
  
  # Check if match_type exists, if not create it
  if (!("match_type" %in% names(player_data))) {
    player_data$match_type <- "Unknown"
    
    # Map game level keys to match types
    # Social matches
    player_data$match_type[player_data$game_level_key == "dsub_social_terrain_beach"] <- "social match"
    player_data$match_type[player_data$game_level_key == "dsub_social_terrain_summer"] <- "social match"
    player_data$match_type[player_data$game_level_key == "dsub_socialspace"] <- "social match"
    
    # PVE matches
    player_data$match_type[player_data$game_level_key == "dsub_region_cheesemines"] <- "pve match"
    player_data$match_type[player_data$game_level_key == "dsub_region_dreadwood"] <- "pve match"
    player_data$match_type[player_data$game_level_key == "dsub_region_towncenter"] <- "pve match"
    
    # PVP matches (Rush)
    pvp_rush_levels <- c("dsub_rush_dreadwood", "dsub_rush_towncenter", "dsub_rush_farm", 
                         "dsub_rush_cheesemines", "dsub_rush_shoreline", "dsub_rush_resort")
    player_data$match_type[player_data$game_level_key %in% pvp_rush_levels] <- "pvp match (rush)"
    
    # PVP matches (Vanquish)
    pvp_vanquish_levels <- c("dsub_tvm_cheesemines", "dsub_tvm_zrefinery", 
                             "dsub_tvm_suburbs", "dsub_tvm_dreadwood")
    player_data$match_type[player_data$game_level_key %in% pvp_vanquish_levels] <- "pvp match (vanquish)"
    
    # PVP matches (Arena)
    player_data$match_type[player_data$game_level_key == "dsub_4v4_battlearena_v2"] <- "pvp match (arena)"
    player_data$match_type[player_data$game_level_key == "dsub_4v4_superarena"] <- "pvp match (arena)"
    
    # Podium celebration
    player_data$match_type[player_data$game_level_key == "dsub_podiumcelebration"] <- "podium celebration"
  }
  
  # Sort by time to get proper sequence
  player_data <- player_data %>%
    arrange(game_start_time, game_end_time)
  
  # Create the plot
  player_data %>%
    dplyr::mutate(row_id = dplyr::row_number()) %>%
    ggplot2::ggplot(
      ggplot2::aes(
        x = start_time_only,
        y = row_id,
        color = match_type
      )
    ) +
    ggplot2::scale_color_manual(
      name = "Match type",
      breaks = c(
        "social match",
        "pve match",
        "pvp match (rush)",
        "pvp match (vanquish)",
        "pvp match (arena)",
        "podium celebration",
        "Unknown"
      ),
      values = c(
        "#2ECC71", "#4E7FE3", "#E74C3C",
        "#F39C12", "#FFEE4B", "#9B59B6", "#CCCCCC"
      )
    ) +
    ggplot2::geom_segment(
      ggplot2::aes(
        xend = end_time_only,
        yend = row_id
      )
    ) +
    ggplot2::scale_x_datetime(
      labels = scales::date_format("%H:%M"),
      breaks = scales::date_breaks("2 hours")
    ) +
    ggplot2::xlab("Time (hours)") +
    ggplot2::ylab("Game matches") +
    ggplot2::theme(
      legend.position = "right",
      axis.text.y = ggplot2::element_blank(),
      axis.ticks.y = ggplot2::element_blank(),
      panel.grid.major = ggplot2::element_blank(),
      panel.grid.minor = ggplot2::element_blank(),
      panel.background = ggplot2::element_blank(),
      axis.line = ggplot2::element_line(colour = "black")
    ) +
    ggplot2::ggtitle(player_label)
}

# Function to scale values to 0-1 range for radar charts
scale_to_01 <- function(x, min_val = min(x, na.rm = TRUE), max_val = max(x, na.rm = TRUE)) {
  if (max_val == min_val) return(rep(0.5, length(x)))
  (x - min_val) / (max_val - min_val)
}

# Function to run ANOVA and extract results
run_anova <- function(dv) {
  formula <- as.formula(paste(dv, "~ player_type"))
  model <- aov(formula, data = pvz_data)
  result <- broom::tidy(model)
  result$eta_squared <- sjstats::eta_sq(model)[["eta.sq"]]
  result$variable <- dv
  return(result)
}

# Function to compute Hopkins statistic for cluster tendency
compute_hopkins <- function(data, n = nrow(data)-1) {
  # Standardize data for clustering
  data_scaled <- scale(data)
  
  # Compute Hopkins statistic
  hopkins_result <- clustertend::hopkins(data_scaled, n = n)
  
  return(hopkins_result)
}

## K-means Clustering

```{r kmeans-clustering}
#| label: kmeans-clustering
#| fig.width: 10
#| fig.height: 6
#| cache: true

# Perform k-means clustering with k=3
set.seed(42) # For reproducibility
k <- 3
kmeans_result <- kmeans(clustering_data_scaled, centers = k, nstart = 25)

# Add cluster assignments to the original data
pvz_df_clustered <- pvz_df_analysis %>%
  mutate(cluster = factor(kmeans_result$cluster))

# Create a function to generate descriptive cluster names
generate_cluster_names <- function(cluster_data) {
  # Calculate cluster centers for key variables
  centers <- cluster_data %>%
    group_by(cluster) %>%
    summarise(
      pvp_ratio = mean(ratio_pvp_hours),
      pve_ratio = mean(ratio_pve_hours),
      social_ratio = mean(ratio_social_hours),
      total_kd = mean(total_kd),
      total_accuracy = mean(total_hit_accuracy),
      hours = mean(hours)
    )
  
  # Assign descriptive names based on the characteristics
  cluster_names <- character(nrow(centers))
  
  for (i in 1:nrow(centers)) {
    # Determine the primary play mode
    mode_ratios <- c(centers$pvp_ratio[i], centers$pve_ratio[i], centers$social_ratio[i])
    mode_names <- c("PvP", "PvE", "Social")
    primary_mode <- mode_names[which.max(mode_ratios)]
    
    # Determine skill level based on KD and accuracy
    skill <- if (centers$total_kd[i] > median(cluster_data$total_kd) && 
                 centers$total_accuracy[i] > median(cluster_data$total_hit_accuracy)) {
      "High-skill"
    } else if (centers$total_kd[i] < median(cluster_data$total_kd) && 
               centers$total_accuracy[i] < median(cluster_data$total_hit_accuracy)) {
      "Casual"
    } else {
      "Mixed-skill"
    }
    
    # Determine playtime intensity
    playtime <- if (centers$hours[i] > median(cluster_data$hours)) {
      "Dedicated"
    } else {
      "Moderate"
    }
    
    # Combine characteristics into a descriptive name
    cluster_names[i] <- paste(playtime, primary_mode, skill, "Players")
  }
  
  return(data.frame(cluster = 1:nrow(centers), player_type = cluster_names))
}

# Generate descriptive names for clusters
cluster_names <- generate_cluster_names(pvz_df_clustered)

# Add descriptive names to the clustered data
pvz_df_clustered <- pvz_df_clustered %>%
  mutate(cluster_num = as.numeric(as.character(cluster))) %>%
  left_join(cluster_names, by = c("cluster_num" = "cluster")) %>%
  select(-cluster_num) %>%
  mutate(cluster = factor(cluster, labels = cluster_names$player_type))

# Save the clustered data for further analysis
write_csv(pvz_df_clustered, here("data/processed/pvz_with_clusters.csv"))
```

# Discussion and Conclusion

This analysis has identified distinct player types in Plants vs. Zombies: Battle for Neighborville based on playstyle metrics, including game mode preference, playtime intensity, and performance measures. The three player clusters exhibit different patterns of well-being and motivation, suggesting that the relationship between gameplay and psychological outcomes varies by play style.

Key findings include:

1. Players show distinct preferences for different game modes (PvP, PvE, and Social), with varying levels of skill and time investment.

2. The identified player types show different relationships between playtime and well-being, suggesting that how players engage with the game matters more than simply how much they play.

3. Motivation factors such as autonomy, competence, and enjoyment vary across player types, indicating different psychological needs being met through gameplay.

4. Game-related well-being tends to be more positive than general well-being across all player types, suggesting that the game provides a positive experience regardless of playstyle.

These findings have implications for game design, suggesting that catering to different playstyles and motivations can enhance player well-being. Future research could explore how these patterns evolve over longer periods and how game design changes might impact different player types.
